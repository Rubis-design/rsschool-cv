import _ from 'lodash';

const getMenCountByYear = (users) => users
  .filter((user) => user.gender === 'male')
  .map((user) => {
    const year = [];
    year.push(user.birthday.slice(0, 4));
    return year;
  })
  .reduce((acc, year) => {
    const count = _.get(acc, year, 0) + 1;
    return { ...acc, [year]: count };
  }, {});

export default getMenCountByYear;
